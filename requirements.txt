# spaCy GPU acceleration — pip picks the right one per platform.
# Windows/Linux: cupy needs CUDA 12.x runtime DLLs (nvrtc, cudart, cublas) at
#                load time. Even with a system NVIDIA CUDA Toolkit installed,
#                cupy fails to locate the DLLs unless CUDA_PATH is set, so we
#                pin the self-contained NVIDIA wheels — pip drops the DLLs into
#                site-packages and cupy finds them automatically. Harmless when
#                a toolkit is also present. ~660 MB total.
#                spacy.prefer_gpu() falls back gracefully when no GPU is found.
# macOS:         thinc-apple-ops routes spaCy matmul through Accelerate.
cupy-cuda12x; sys_platform != "darwin"
nvidia-cuda-nvrtc-cu12; sys_platform != "darwin"
nvidia-cuda-runtime-cu12; sys_platform != "darwin"
nvidia-cublas-cu12; sys_platform != "darwin"
thinc-apple-ops; sys_platform == "darwin"

# Default PyTorch wheels include Apple Silicon MPS support on macOS.
# NVIDIA CUDA machines must use a wheel matching their driver/runtime.
# AMD ROCm users should install their ROCm wheel first, then install
# requirements-amd.txt instead of this NVIDIA-oriented requirements file.
torch
